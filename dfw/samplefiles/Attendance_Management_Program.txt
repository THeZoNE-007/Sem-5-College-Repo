# To check whether student Present or Absent
def update_attendance(details):
    if details in students:
        student = students[details]
    else:
        for student in students.values():
            if student["Roll Number"] == details:
                break
        else:
            print("Student not found.")
            return

    present = input(f"{details} is present? (Y/N) ").upper()
    if present == "Y":
        student["Attendance"] += 1
        student["Attendance Percentage"] = (student["Attendance"] / 365) * 100
        print(f"{details} attendance updated. Current attendance: {student['Attendance']}, Attendance percentage: {student['Attendance Percentage']:.2f}%")
    else:
        print(f"{details} absence recorded.")

# To print the particular student details when name entered
def print_student_details(details):
    if details in students:
        student = students[details]
        print(f"Name: {details}")
        print(f"Roll Number: {student['Roll Number']}")
        print(f"Attendance: {student['Attendance']}")
        print(f"Attendance Percentage: {student['Attendance Percentage']:.2f}%")
        print(f"Admission Number: {student['Admission Number']}")
    else:
        for student in students.values():
            if student["Roll Number"] == details:
                print(f"Name: {list(students.keys())[list(students.values()).index(student)]}")
                print(f"Roll Number: {student['Roll Number']}")
                print(f"Attendance: {student['Attendance']}")
                print(f"Attendance Percentage: {student['Attendance Percentage']:.2f}%")
                print(f"Admission Number: {student['Admission Number']}")
                break
        else:
            print("Student not found.")

#To modify student details
def modify_student_details(details):
    if details in students:
        student = students[details]
        while True:
            print("\nModify Student Details:")
            print("1. Attendance")
            print("2. Attendance Percentage")
            print("3. Admission Number")
            print("4. Exit")
            choice = input("Choose an option: ")
            if choice == "1":
                student["Attendance"] = int(input("Enter new attendance: "))
                student["Attendance Percentage"] = (student["Attendance"] / 365) * 100
                print(f"{details} attendance updated. Current attendance: {student['Attendance']}, Attendance percentage: {student['Attendance Percentage']:.2f}%")
            elif choice == "2":
                student["Attendance Percentage"] = (int(input("Enter new attendance: ")) / 365) * 100
                print(f"{details} attendance percentage updated. Current attendance: {student['Attendance']}, Attendance percentage: {student['Attendance Percentage']:.2f}%")
            elif choice == "3":
                student["Admission Number"] = int(input("Enter new admission number: "))
                print(f"{details} admission number updated. Admission number: {student['Admission Number']}")
            elif choice == "4":
                print(f"{details} details updated.")
                break
            else:
                print("Invalid choice. Please try again.")
    else:
        print("Student not found.")
# Running the main function to choose between T/S/EXIT
def main():
    while True:
        user_type = input("Are you a teacher or student? (T/S) \nType EXIT to end program: ").upper()
        if user_type == "T":
            print("All student details:")
            for name in students:
                print(name)
            details = input("Enter student name: ")
            print_student_details(details)
            modify = input("Do you want to modify the details? (Y/N) ").upper()
            if modify == "Y":
                modify_student_details(details)
            elif modify == "N":
                print("Program terminated.")
                break
        elif user_type == "S":
            details = input("Enter your name: ")
            print_student_details(details)
            update_attendance(details)
        elif user_type == "EXIT":
            print("Program terminated.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
